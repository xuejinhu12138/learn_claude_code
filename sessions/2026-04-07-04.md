## 关键代码结构

```typescript
// src/agent.ts - Agentic Loop 核心
while (isContinue) {
    const { text, stop_reason, tool_use } = await sendMessage(getHistory());
    addMessage(createAssistantMessage(text, tool_use));
    switch (stop_reason) {
        case 'tool_calls':
            await Promise.all(
                tool_use.map(async ({ id, function: { name, arguments: ag } }) => {
                    const tool = toolRegistry.get(name);
                    try {
                        const input = JSON.parse(ag || "{}");
                        if (!tool) throw new Error(`Tool ${name} not found`);
                        const result = await tool.call(input);
                        addMessage(createToolMessage(id, result));
                    } catch (err) {
                        addMessage(createToolMessage(id, `Error: ${err instanceof Error ? err.message : String(err)}`));
                    }
                })
            );
            turns++;
            if (turns > MAX_TURNS) { isContinue = false; }
            break;
        default:
            isContinue = false;
    }
}
```
